1. Cayley-Hamilton Theorem

syms A s
A = [7 2 -2; -6 -1 2; 6 2 -1];
I = eye(3);
charpoly(A, s) == 0

verification = A^3 - 5*A^2 + 7*A - 3*I

format rational
inv_A = (1/3)*(A^2 - 5*A + 7*I)

A_power4 = 5*A^3 - 7*A^2 + 3*A

Output: Verifies the Cayley-Hamilton theorem, calculates inverse and power.

4. Rolle's Theorem

syms x
f(x) = (x - 1)^2 * (x - 3);
df = diff(f, x);
critical_points = solve(df == 0);
disp('Critical Points:');
disp(critical_points);

Output: Critical points within interval (1, 3).

5. Mean Value Theorem

syms x
f(x) = x^2;
a = 0; b = 1;
m = (f(b) - f(a)) / (b - a);
c = solve(diff(f, x) == m, x);
disp('Point c satisfying Mean Value Theorem:');
disp(c);

Output: Value of c satisfying the theorem.

7. Maxima and Minima

syms x
f(x) = x^3 - 6*x^2 + 9*x + 1;
df = diff(f, x);
critical_points = solve(df == 0);
second_derivative = diff(f, x, 2);
disp('Critical Points and Nature:');
for i = 1:length(critical_points)
    if subs(second_derivative, x, critical_points(i)) > 0
        disp(['Point ', char(critical_points(i)), ' is a Minimum']);
    else
        disp(['Point ', char(critical_points(i)), ' is a Maximum']);
    end
end

Output: Nature of critical points (minima/maxima).

18. Analytic Functions

syms x y
u = x^2 - y^2;
v = 2*x*y;
c1 = diff(u, x) - diff(v, y);
c2 = diff(u, y) + diff(v, x);
if c1 == 0 && c2 == 0
    disp('Function is analytic');
else
    disp('Function is not analytic');
end

Output: Analyticity test result.
