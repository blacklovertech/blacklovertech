1. Cayley-Hamilton Theorem

syms A s
A = [7 2 -2; -6 -1 2; 6 2 -1];
I = eye(3); % Identity matrix
charpoly(A, s) == 0

verification = A^3 - 5*A^2 + 7*A - 3*I

format rational
inv_A = (1/3)*(A^2 - 5*A + 7*I)

A_power4 = 5*A^3 - 7*A^2 + 3*A

Output: Verification = Zero matrix

2. Diagonalization of Matrices

disp('Eigenvalues:');
disp(D);
disp('Eigenvectors:');

Output: Eigenvalues = [-2, 3, 6]; Eigenvectors matrix displayed.

4. Rolle's Theorem

syms x
f(x) = (x - 1)^2 * (x - 3);
df = diff(f, x);
critical_points = solve(df == 0);
disp('Critical Points:');
disp(critical_points);

Output: Critical points within the interval displayed.

5. Mean Value Theorem

syms x
f(x) = x^2;
a = 0; b = 1;
m = (f(b) - f(a)) / (b - a);
c = solve(diff(f, x) == m, x);
disp('Point c satisfying Mean Value Theorem:');
disp(c);

Output: Point c within the interval displayed.

7. Maxima and Minima

syms x
f(x) = x^3 - 6*x^2 + 9*x + 1;
df = diff(f, x);
critical_points = solve(df == 0);
second_derivative = diff(f, x, 2);
disp('Critical Points and Nature:');
for i = 1:length(critical_points)
    if subs(second_derivative, x, critical_points(i)) > 0
        disp(['Point ', char(critical_points(i)), ' is a Minimum']);
    else
        disp(['Point ', char(critical_points(i)), ' is a Maximum']);
    end
end

Output: Critical points identified as minima or maxima.

11. Divergence

syms x y z
F = [x^2*y, -2*x*z, 2*y*z];
div_F = divergence(F);
disp('Divergence:');
disp(div_F);

Output: Divergence = 2xy - 2z.
